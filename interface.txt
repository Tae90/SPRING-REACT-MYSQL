- request
{
    *email : String,
    *password : String
}

- response

성공

{
    code: "Su",
    message: "Sucess."
    token: "jwt...",
    expiredDate: 123456789
}

    code: "SF",

}

{
    code: "DE",
    message: "Database Error."
}
